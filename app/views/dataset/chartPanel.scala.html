@import util.{ChartSpec, PieChartSpec, ColumnChartSpec} @import views.html.chart._ @(chartSpecs : Iterable[ChartSpec])
@chartSpecs.map { chartSpec =>
@{chartSpec match { case PieChartSpec(title, showLabels, showLegend, data) => highcharts.pieChart(title, title.replaceAll(" ", "-") + "Chart", showLabels, showLegend, data) case ColumnChartSpec(title, data) => highcharts.columnChart(title, title.replaceAll(" ", "-") + "Chart", "Count", data) case _ => throw new IllegalArgumentException(s"Chart spec $chartSpec is unknown.") }}
}
